Future<void> sendNotif(String userId, Product product, Family family) async {
  final apiInstance = os_api.DefaultApi();
  const appId = "4344420b-6f4f-41af-9763-ec2c852e3c5b";
  const key = "YTlmOTM0MDctYzA1MC00MWY1LWJlOTMtZTAyYjY3N2Y4NTRj";

  try {
    // authentictae api
    apiInstance.apiClient.addDefaultHeader("Authorization", "Basic $key");

    final notification = os_api.Notification(
        appId: appId,
        includeExternalUserIds: [userId],
        isAndroid: true,
        channelForExternalUserIds: 'push',
        isIos: true,
        contents: os_api.StringMap(
          en: "A new item was assigned to you.\nProduct: ${product.name}, Category: ${product.categoryName!.titleCase}.\nClick to view",
        ),
        headings: os_api.StringMap(
          en: "New Item Assigned",
        ),
        data: {"product": product, "family": family});

    // send notification
    await apiInstance.createNotification(notification);
  } catch (e) {
    debugPrint(e.toString());
  }
}
